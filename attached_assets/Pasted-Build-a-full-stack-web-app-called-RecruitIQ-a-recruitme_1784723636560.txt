DATABASE SCHEMA (create exactly this, do not add extra tables or change field names):

-- ============================================
-- CANDIDATES: core info about each person
-- ============================================
CREATE TABLE candidates (
    id SERIAL PRIMARY KEY,
    name TEXT NOT NULL,
    email TEXT,
    phone TEXT,
    resume_text TEXT,              -- raw text extracted from resume (for search/reference)
    skills TEXT,                   -- comma-separated or freeform, kept simple for MVP
    source TEXT CHECK (source IN ('resume', 'chat', 'voice', 'mixed')),
    created_at TIMESTAMP DEFAULT NOW(),
    updated_at TIMESTAMP DEFAULT NOW()
);

-- Speeds up duplicate-candidate lookups by email/phone before inserting a new person
CREATE INDEX idx_candidates_email ON candidates(email);
CREATE INDEX idx_candidates_phone ON candidates(phone);

-- ============================================
-- CANDIDATE_PREFERENCES: what the AI extracts
-- One row per candidate (1-to-1 with candidates)
-- ============================================
CREATE TABLE candidate_preferences (
    id SERIAL PRIMARY KEY,
    candidate_id INTEGER NOT NULL REFERENCES candidates(id) ON DELETE CASCADE,
    desired_role TEXT,                  -- raw text as extracted, e.g. "People Operations"
    desired_role_category TEXT,         -- standardized bucket, e.g. "HR" — used for searching/matching
    salary_expectation TEXT,            -- kept as text since ranges like "8-10 LPA" are common
    notice_period TEXT,                 -- e.g. "immediate", "30 days", "60 days"
    wfh_preference TEXT CHECK (wfh_preference IN ('remote', 'hybrid', 'onsite', 'no_preference')),
    relocation_willingness BOOLEAN,
    shift_preference TEXT,              -- e.g. "day", "night", "flexible"
    career_interests TEXT,              -- freeform notes on long-term interests
    last_verified_at TIMESTAMP,         -- last time someone actually confirmed candidate is still available
    updated_at TIMESTAMP DEFAULT NOW()
);

CREATE INDEX idx_preferences_role_category ON candidate_preferences(desired_role_category);

-- ============================================
-- JOB_OPENINGS: the roles you're hiring for
-- ============================================
CREATE TABLE job_openings (
    id SERIAL PRIMARY KEY,
    title TEXT NOT NULL,
    role_category TEXT,            -- standardized bucket, matches desired_role_category above
    requirements TEXT,
    created_at TIMESTAMP DEFAULT NOW()
);

-- ============================================
-- CANDIDATE_JOB_STATUS: the junction table
-- Tracks each candidate's status PER job opening,
-- so the same person can be "rejected" for one role
-- and "open" for another, over time.
-- ============================================
CREATE TABLE candidate_job_status (
    id SERIAL PRIMARY KEY,
    candidate_id INTEGER NOT NULL REFERENCES candidates(id) ON DELETE CASCADE,
    job_id INTEGER NOT NULL REFERENCES job_openings(id) ON DELETE CASCADE,
    status TEXT CHECK (status IN ('matched', 'rejected', 'open_for_future', 'interested', 'contacted')) NOT NULL,
    updated_by TEXT,               -- name of whoever last changed this status (Tashmeet / friend 1 / friend 2)
    updated_at TIMESTAMP DEFAULT NOW(),
    UNIQUE (candidate_id, job_id)
);

CREATE INDEX idx_status_lookup ON candidate_job_status(status, job_id);
